import java.util.*; import java.io.*; import java.lang.Math; public class Main { public static int numberOfEqualStrings(int N, String S) { int result = 0; if (N <= 1) return 0; char prev = S.charAt(0); int currentRunLength = 1; for (int i = 1; i < N; i++) { if (S.charAt(i) == prev) { currentRunLength++; } else { result += currentRunLength * (currentRunLength - 1) / 2; prev = S.charAt(i); currentRunLength = 1; } } // Add the last run result += currentRunLength * (currentRunLength - 1) / 2; return result; } public static void main(String[] args) { Scanner sc = new Scanner(System.in); int N = sc.nextInt(); String S = sc.next(); sc.close(); System.out.print(numberOfEqualStrings(N, S)); } }
