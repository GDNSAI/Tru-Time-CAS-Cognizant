const express = require('express'); const bodyParser = require('body-parser'); const Excel = require('exceljs'); const moment = require('moment'); const fs = require('fs'); const path = require('path'); const app = express(); const port = 3000; app.use(bodyParser.json()); app.use(bodyParser.urlencoded({ extended: true })); app.use(express.static('public')); const excelFilePath = 'C:/Users/2308397/PilotProject-Ally/attandence.xlsx'; async function createExcelFile() { const workbook = new Excel.Workbook(); const worksheet = workbook.addWorksheet('Data'); const currentDate = moment(); const datesToUse = getMonthDates(currentDate); const headers = [ 'Associate ID', 'Associate Name', 'Status', 'Productivity', 'Project Name', 'Allocation', 'Started On', 'Onshore/Offshore', 'Location', ...datesToUse ]; worksheet.addRow(headers); await workbook.xlsx.writeFile(excelFilePath); console.log('Excel file created with headers'); } function getMonthDates(date) { const monthStart = date.clone().startOf('month'); const monthEnd = date.clone().endOf('month'); const dates = []; let currentDate = monthStart.clone(); while (currentDate.isSameOrBefore(monthEnd)) { dates.push(currentDate.format('DD-MMM')); currentDate.add(1, 'day'); } return dates; } // Create new Excel file if it doesn't exist or month has changed async function initializeExcelFile() { if (!fs.existsSync(excelFilePath)) { await createExcelFile(); return; } const workbook = new Excel.Workbook(); await workbook.xlsx.readFile(excelFilePath); const worksheet = workbook.getWorksheet('Data'); const existingDates = worksheet.getRow(1).values .slice(9) // Skip first 9 columns .filter(d => d); const currentDates = getMonthDates(moment()); if (existingDates.join(',') !== currentDates.join(',')) { await createExcelFile(); } } app.post('/add-data', async (req, res) => { try { const { associate_id, associate_name, status, productivity, project_name, allocation, started_on, onshore_offshore, location, leave_dates } = req.body; const workbook = new Excel.Workbook(); await workbook.xlsx.readFile(excelFilePath); const worksheet = workbook.getWorksheet('Data'); const currentDates = getMonthDates(moment()); const leaveSet = new Set(leave_dates); const rowData = [ associate_id, associate_name, status, productivity, project_name, allocation, moment(started_on).format('DD-MMM-YYYY'), onshore_offshore, location ]; currentDates.forEach(date => { rowData.push(leaveSet.has(date) ? 'Personal Leave' : 'Present'); }); worksheet.addRow(rowData); await workbook.xlsx.writeFile(excelFilePath); res.send('Data added successfully'); } catch (error) { console.error('Error:', error); res.status(500).send('Error adding data'); } }); initializeExcelFile().then(() => { app.listen(port, () => { console.log(`Server running at http://localhost:${port}`); }); });
