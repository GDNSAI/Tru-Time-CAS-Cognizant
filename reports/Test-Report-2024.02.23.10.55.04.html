const express = require('express'); const bodyParser = require('body-parser'); const XLSX = require('xlsx'); const fs = require('fs'); const app = express(); app.use(bodyParser.urlencoded({ extended: true })); // Generate date headers for current month function getDateHeaders() { const now = new Date(); const year = now.getFullYear(); const month = now.getMonth(); const daysInMonth = new Date(year, month + 1, 0).getDate(); return Array.from({ length: daysInMonth }, (_, i) => { const date = new Date(year, month, i + 1); return date.toLocaleDateString('en-GB', { day: '2-digit', month: 'short' }).replace(/ /g, '-'); }); } app.post('/submit', (req, res) => { const formData = req.body; const filePath = 'attendance.xlsx'; // Validate required fields const requiredFields = ['associateName', 'associateId', 'status', 'projectId', 'projectName', 'allocationStartDate', 'onshoreOffshore', 'location', 'leaveDates', 'leaveType']; if (requiredFields.some(field => !formData[field])) { return res.status(400).json({ error: 'All fields are required' }); } // Process leave dates const currentDateHeaders = getDateHeaders(); const currentMonth = new Date().getMonth(); const currentYear = new Date().getFullYear(); const leaveDates = formData.leaveDates.split(',').map(dateStr => { const [year, month, day] = dateStr.split('-').map(Number); const date = new Date(year, month - 1, day); if (date.getMonth() === currentMonth && date.getFullYear() === currentYear) { return date.toLocaleDateString('en-GB', { day: '2-digit', month: 'short' }).replace(/ /g, '-'); } return null; }).filter(d => d !== null); // Prepare new row data const defaultStatus = formData.status === 'Active' ? 'BLD' : 'UBL'; const dateColumns = {}; currentDateHeaders.forEach(header => { dateColumns[header] = leaveDates.includes(header) ? formData.leaveType : defaultStatus; }); const newRow = { 'Associate Name': formData.associateName, 'Associate ID': formData.associateId, 'Status': formData.status, 'Project ID': formData.projectId, 'Project Name': formData.projectName, 'Allocation Start Date': formData.allocationStartDate, 'Onshore/Offshore': formData.onshoreOffshore, 'Location': formData.location, ...dateColumns }; // Read and update existing data let data = []; if (fs.existsSync(filePath)) { const workbook = XLSX.readFile(filePath); const worksheet = workbook.Sheets['Sheet1']; data = XLSX.utils.sheet_to_json(worksheet); // Update existing rows to current date format data = data.map(row => { const rowStatus = row['Status'] || 'Active'; const rowDefault = rowStatus === 'Active' ? 'BLD' : 'UBL'; const updatedRow = { ...row, ...currentDateHeaders.reduce((acc, header) => { acc[header] = row[header] || rowDefault; return acc; }, {}) }; // Remove old date columns Object.keys(updatedRow).forEach(key => { if (![...requiredFields.map(f => f.replace(/([A-Z])/g, ' $1').trim()), ...currentDateHeaders].includes(key)) { delete updatedRow[key]; } }); return updatedRow; }); } // Update or add new entry const existingIndex = data.findIndex(row => row['Associate ID'] === formData.associateId); if (existingIndex !== -1) { data[existingIndex] = newRow; } else { data.push(newRow); } // Create new workbook const workbook = XLSX.utils.book_new(); const worksheet = XLSX.utils.json_to_sheet(data); XLSX.utils.book_append_sheet(workbook, worksheet, 'Sheet1'); XLSX.writeFile(workbook, filePath); res.json({ message: 'Data saved successfully!' }); }); app.get('/fetch-data', (req, res) => { // ... existing fetch-data implementation ... }); app.listen(3000, () => console.log('Server running on port 3000')); document.addEventListener('DOMContentLoaded', function() { flatpickr("#allocationStartDate", { dateFormat: "Y-m-d" }); const leavePicker = flatpickr("#leaveDates", { mode: "multiple", dateFormat: "Y-m-d", onChange: function(selectedDates) { const formattedDates = selectedDates.map(date => date.toLocaleDateString('en-GB', { day: '2-digit', month: 'short' }) .replace(/ /g, '-') ); document.getElementById('leaveDates').value = formattedDates.join(','); } }); });
