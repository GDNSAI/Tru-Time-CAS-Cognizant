import java.util.Scanner; public class Main { public static int countDigitOne(int N) { if (N < 0 || N > 100) { throw new IllegalArgumentException("The Number is Not Valid"); } int count = 0; for (int i = 0; i <= N; i++) { String numStr = Integer.toString(i); for (int j = 0; j < numStr.length(); j++) { if (numStr.charAt(j) == '1') { count++; } } } return count; } public static void main(String[] args) { Scanner sc = new Scanner(System.in); int N = sc.nextInt(); sc.close(); try { System.out.println(countDigitOne(N)); } catch (Exception e) { System.out.println(e.getMessage()); } } }
