Attendance Management Form

const express = require('express'); const bodyParser = require('body-parser'); const ExcelJS = require('exceljs'); const fs = require('fs'); const app = express(); app.use(bodyParser.urlencoded({ extended: true })); app.use(express.static('public')); app.post('/submit', async (req, res) => { const { associateName, associateId, status, projectId, projectName, allocationStartDate, onshoreOffshore, location, leaveDates, leaveType } = req.body; // Validate all required fields if (!associateName || !associateId || !status || !projectId || !projectName || !allocationStartDate || !onshoreOffshore || !location || !leaveDates || !leaveType) { return res.status(400).json({ error: 'All fields are required' }); } const filePath = 'attendance.xlsx'; const headers = ['Associate Name', 'Associate ID', 'Status', 'Project ID', 'Project Name', 'Allocation Start Date', 'Onshore/Offshore', 'Location', 'Leave Dates', 'Leave Type']; let data = []; const workbook = new ExcelJS.Workbook(); if (fs.existsSync(filePath)) { await workbook.xlsx.readFile(filePath); } else { const worksheet = workbook.addWorksheet('Sheet1'); worksheet.columns = headers.map(header => ({ header, key: header })); } const worksheet = workbook.getWorksheet('Sheet1'); data = worksheet.getSheetValues().slice(1).map(row => { return headers.reduce((acc, header, index) => { acc[header] = row[index + 1]; return acc; }, {}); }); // Check if the Associate ID already exists const existingIndex = data.findIndex(row => row['Associate ID'] === associateId); // Prepare the new row data const newRow = { 'Associate Name': associateName, 'Associate ID': associateId, 'Status': status, 'Project ID': projectId, 'Project Name': projectName, 'Allocation Start Date': allocationStartDate, 'Onshore/Offshore': onshoreOffshore, 'Location': location, 'Leave Dates': leaveDates, 'Leave Type': leaveType }; // Update or add the new row if (existingIndex !== -1) { worksheet.getRow(existingIndex + 2).values = Object.values(newRow); } else { worksheet.addRow(newRow); } await workbook.xlsx.writeFile(filePath); res.json({ message: 'Data saved successfully!' }); }); app.listen(3000, () => console.log('Server running on port 3000'));
