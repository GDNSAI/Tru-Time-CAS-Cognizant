const OpenAI = require('openai'); const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY }); app.post('/ai-command', async (req, res) => { const { prompt } = req.body; try { const completion = await openai.chat.completions.create({ messages: [{ role: "system", content: `You are an HR assistant working with employee data. Available actions: 1. Search employees: fields include associateId, name, location, status 2. Update records: fields from the form 3. Data analysis: show statistics Respond in JSON format with: { "action": "search|update|stats", "message": "human-readable response", "data": {} // parameters for action }` }, { role: "user", content: prompt }], model: "gpt-3.5-turbo", response_format: { type: "json_object" } }); const response = JSON.parse(completion.choices[0].message.content); // Process the AI command if (response.action === 'search') { const results = await searchRecords(response.data); response.message += `\nFound ${results.length} records:`; response.data = results; } else if (response.action === 'update') { await handleUpdate(response.data); } res.json(response); } catch (error) { res.status(500).json({ error: error.message }); } }); async function searchRecords(filters) { const workbook = XLSX.readFile('attendance.xlsx'); const data = XLSX.utils.sheet_to_json(workbook.Sheets['Sheet2']); return data.filter(record => { return Object.entries(filters).every(([key, value]) => { return record[key].toString().toLowerCase() === value.toLowerCase(); }); }); } async function handleUpdate(data) { // Validate and update logic similar to /submit endpoint const filePath = 'attendance.xlsx'; const workbook = XLSX.readFile(filePath); const worksheet = workbook.Sheets['Sheet2']; const records = XLSX.utils.sheet_to_json(worksheet); const index = records.findIndex(r => r['Associate ID'] === data.associateId && r['Status'] === data.status ); if (index === -1) throw new Error('Record not found'); // Update record Object.entries(data).forEach(([key, value]) => { records[index][key] = value; }); // Save updated file const newWorksheet = XLSX.utils.json_to_sheet(records); const newWorkbook = XLSX.utils.book_new(); XLSX.utils.book_append_sheet(newWorkbook, newWorksheet, 'Sheet2'); XLSX.writeFile(newWorkbook, filePath); }
