if (!onshoreOffshore) missingFields.push('onshoreOffshore'); if (!location) missingFields.push('location'); if (!leaveDates) missingFields.push('leaveDates'); if (!leaveType) missingFields.push('leaveType'); if (missingFields.length > 0) { console.log('Missing fields:', missingFields); return res.status(400).json({ error: "Missing required fields", missingFields }); } const workbook = new ExcelJS.Workbook(); const filePath = path.join(__dirname, 'C:/Users/2308397/PilotProject-Ally/PilotProject-Ally.xlsx'); try { await workbook.xlsx.readFile(filePath); } catch (error) { const worksheet = workbook.addWorksheet('Attendance'); worksheet.columns = [ { header: 'ID', key: 'id' }, { header: 'Timestamp', key: 'timestamp' }, { header: 'Associate ID', key: 'associateId' }, { header: 'Associate Name', key: 'associateName' }, { header: 'Status', key: 'status' }, { header: 'Project ID', key: 'projectId' }, { header: 'Project Name', key: 'projectName' }, { header: 'Allocation Start Date', key: 'allocationStartDate' }, { header: 'Onshore/Offshore', key: 'onshoreOffshore' }, { header: 'Location', key: 'location' }, { header: 'Leave Dates', key: 'leaveDates' }, { header: 'Leave Type', key: 'leaveType' } ]; } const worksheet = workbook.getWorksheet('Attendance'); const newRow = { id: worksheet.rowCount, timestamp: new Date().toISOString(), associateId, associateName, status, projectId, projectName, allocationStartDate, onshoreOffshore, location, leaveDates, leaveType }; worksheet.addRow(newRow); await workbook.xlsx.writeFile(filePath); res.send('Data saved successfully!'); }); app.listen(port, () => { console.log(`Server running at http://localhost:${port}`); });
